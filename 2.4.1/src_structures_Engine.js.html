    const Constants = require('../../utils/Constants.js');

/**
 * Represents a Dolphin Engine
 * @class
 */
class Engine {
    /**
     * @param {EngineOptions} options Options for a Dolphin Engine
     */
    constructor(options = Constants.EngineOptions) {
        this.options = options;
    }

    execute(client, ...args) {
        this.client = client;
        if (!this.options.timeout) return this.run(...args);

        setInterval(() => this.run(...args), this.options.timeout * 60000);
    }
}

module.exports = Engine;
